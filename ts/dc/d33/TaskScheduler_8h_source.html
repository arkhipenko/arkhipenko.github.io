504 if ( iCount) { // do not update the status request if it was already completed
505 if (iCount > 0) --iCount;
506 if ( (iStatus = aStatus) < 0 ) iCount = 0; // if an error is reported, the status is requested to be completed immediately

539 return enable();

1043 void Scheduler::addTask(Task& aTask) {
1044// If task already belongs to a scheduler, we should not be adding
1045// it to this scheduler. It should be deleted from the other scheduler first.
1046 if (aTask.iScheduler != NULL)
1047 return;
1048
1049 iEnabled = false;
1050
1051 aTask.iScheduler = this;
1052// First task situation:
1053 if (iFirst == NULL) {
1054 iFirst = &aTask;
1055 aTask.iPrev = NULL;
1056 }
1057 else {
1058// This task gets linked back to the previous last one
1059 aTask.iPrev = iLast;
1060 iLast->iNext = &aTask;
1061 }
1062// "Previous" last task gets linked to this one - as this one becomes the last one
1063 aTask.iNext = NULL;
1064 iLast = &aTask;
1065
1066 iEnabled = true;
1067}

1072void Scheduler::deleteTask(Task& aTask) {
1073// Can only delete own tasks
1074 if (aTask.iScheduler != this)
1075 return;
1076
1077 iEnabled = false;
1078
1079 aTask.iScheduler = NULL;
1080 if (aTask.iPrev == NULL) {
1081 if (aTask.iNext == NULL) {
1082 iFirst = NULL;
1083 iLast = NULL;
1084 iEnabled = true;
1085 return;
1086 }
1087 else {
1088 aTask.iNext->iPrev = NULL;
1089 iFirst = aTask.iNext;
1090 aTask.iNext = NULL;
1091 iEnabled = true;
1092 return;
1093 }
1094 }
1095
1096 if (aTask.iNext == NULL) {
1097 aTask.iPrev->iNext = NULL;
1098 iLast = aTask.iPrev;
1099 aTask.iPrev = NULL;
1100 iEnabled = true;
1101 return;
1102 }
1103
1104 aTask.iPrev->iNext = aTask.iNext;
1105 aTask.iNext->iPrev = aTask.iPrev;
1106 aTask.iPrev = NULL;
1107 aTask.iNext = NULL;
1108
1109 iEnabled = true;
1110}

1528 bool idleRun = true;
1529 unsigned long m, i; // millis, interval;
1530
1531#if defined(_TASK_TIMECRITICAL)
1532 unsigned long tPassStart;
1533 unsigned long tTaskStart, tTaskFinish;
1534#endif // _TASK_TIMECRITICAL
1535
1536#ifdef _TASK_SLEEP_ON_IDLE_RUN
1537 unsigned long tFinish;
1538 unsigned long tStart;
1539#ifdef _TASK_TIMECRITICAL
1540 unsigned long tIdleStart = 0;
1541#endif // _TASK_TIMECRITICAL
1542#endif // _TASK_SLEEP_ON_IDLE_RUN
1543
1544
1545 Task *nextTask; // support for deleting the task in the onDisable method
1546 iCurrent = iFirst;
1547
1548 iActiveTasks = 0;
1549 iTotalTasks = 0;
1550 iInvokedTasks = 0;
1551
1552#ifdef _TASK_PRIORITY
1553 // If lower priority scheduler does not have a single task in the chain
1554 // the higher priority scheduler still has to have a chance to run
1555 if (!iCurrent && iHighPriority) iHighPriority->execute();
1556 iCurrentScheduler = this;
1557#endif // _TASK_PRIORITY
1558
1559 // each scheduled is enabled/disabled individually, so check iEnabled only
1560 // after the higher priority scheduler has been invoked.
1561 if ( !iEnabled ) return true; // consider this to be an idle run
1562
1563#ifdef _TASK_THREAD_SAFE
1564 // Process external requests for task updates
1565 // The requests are processed in bulk, in the order they were received
1566 processRequests();
1567#endif
1568
1569#ifdef _TASK_SLEEP_ON_IDLE_RUN
1570 // scheduling pass starts
1571 tStart = _task_micros();
1572#endif
1573
1574#ifdef _TASK_TICKLESS
1575 unsigned long nr = UINT32_MAX; // we do not know yet if we can tell when next run will be
1576 unsigned int nrd = _TASK_NEXTRUN_UNDEFINED;
1577#endif
1578
1579
1580 while (!iPaused && iCurrent) {
1581
1582#ifdef _TASK_THREAD_SAFE
1583 // Process external requests for task on every pass
1584 // The requests are processed in bulk, in the order they were received
1585 // This should emulate a one-thread behaviour
1586 processRequests();
1587#endif
1588
1589 iTotalTasks++;
1590
1591#if defined(_TASK_TIMECRITICAL)
1592 tPassStart = _task_micros();
1593 tTaskStart = tTaskFinish = 0;
1594#endif // _TASK_TIMECRITICAL
1595
1596#ifdef _TASK_PRIORITY
1597 // If scheduler for higher priority tasks is set, it's entire chain is executed on every pass of the base scheduler
1598 if (iHighPriority) idleRun = iHighPriority->execute() && idleRun;
1599 iCurrentScheduler = this;
1600#endif // _TASK_PRIORITY
1601 nextTask = iCurrent->iNext;
1602 do {
1603 if ( iCurrent->iStatus.enabled ) {
1604 iActiveTasks++;
1605
1606#ifdef _TASK_WDT_IDS
1607 // For each task the control points are initialized to avoid confusion because of carry-over:
1608 iCurrent->iControlPoint = 0;
1609#endif // _TASK_WDT_IDS
1610
1611 // Disable task on last iteration:
1612 if (iCurrent->iIterations == 0) {
1613 iCurrent->disable();
1614#ifdef _TASK_SELF_DESTRUCT
1615 if ( iCurrent->iStatus.sd_request ) delete iCurrent;
1616#endif // #ifdef _TASK_SELF_DESTRUCT
1617 break;
1618 }
1619 m = __TASK_TIME_FUNCTION();
1620 i = iCurrent->iInterval;
1621
1622#ifdef _TASK_TIMEOUT
1623 // Disable task on a timeout
1624 if ( iCurrent->iTimeout && (m - iCurrent->iStarttime > iCurrent->iTimeout) ) {
1625 iCurrent->iStatus.timeout = true;
1626 iCurrent->disable();
1627#ifdef _TASK_SELF_DESTRUCT
1628 if ( iCurrent->iStatus.sd_request ) delete iCurrent;
1629#endif // #ifdef _TASK_SELF_DESTRUCT
1630 break;
1631 }
1632#endif // _TASK_TIMEOUT
1633
1634#ifdef _TASK_STATUS_REQUEST
1635 // If StatusRequest object was provided, and still pending, and task is waiting, this task should not run
1636 // Otherwise, continue with execution as usual. Tasks waiting to StatusRequest need to be rescheduled according to
1637 // how they were placed into waiting state (waitFor or waitForDelayed)
1638 if ( iCurrent->iStatus.waiting ) {
1639
1640#ifdef _TASK_TICKLESS
1641 // if there is a task waiting on a status request we are obligated to run continously
1642 // because event can trigger at any point at time.
1643 nrd |= _TASK_NEXTRUN_IMMEDIATE; // immediate
1644#endif
1645
1646#ifdef _TASK_TIMEOUT
1647 StatusRequest *sr = iCurrent->iStatusRequest;
1648 if ( sr->iTimeout && (m - sr->iStarttime > sr->iTimeout) ) {
1649 sr->signalComplete(TASK_SR_TIMEOUT);
1650 }
1651#endif // _TASK_TIMEOUT
1652 if ( (iCurrent->iStatusRequest)->pending() ) break;
1653 if ( (iCurrent->iStatusRequest)->iStatus == TASK_SR_ABORT ) {
1654 iCurrent->abort();
1655 break;
1656 }
1657 if (iCurrent->iStatus.waiting == _TASK_SR_NODELAY) {
1658 iCurrent->iDelay = i;
1659 iCurrent->iPreviousMillis = m - i;
1660 }
1661 else {
1662 iCurrent->iPreviousMillis = m;
1663 }
1664 iCurrent->iStatus.waiting = 0;
1665 }
1666#endif // _TASK_STATUS_REQUEST
1667
1668 // this is the main scheduling decision point
1669 // if the interval between current time and previous invokation time is less than the current delay - task should NOT be activated yet.
1670 // this is millis-rollover-safe way of scheduling
1671 if ( m - iCurrent->iPreviousMillis < iCurrent->iDelay ) {
1672#ifdef _TASK_TICKLESS
1673 // catch the reamining time until invocation as next time this should run
1674 // this does not handle millis rollover well - so for the rollover situation (once every 47 days)
1675 // we will require immediate execution
1676 unsigned long nextrun = iCurrent->iDelay + iCurrent->iPreviousMillis;
1677 // nextrun should be after current millis() (except rollover)
1678 // nextrun should be sooner than previously determined
1679 if ( nextrun > m && nextrun < nr ) {
1680 nr = nextrun;
1681 nrd |= _TASK_NEXTRUN_TIMED; // next run timed
1682 }
1683#endif // _TASK_TICKLESS
1684 break;
1685 }
1686
1687
1688#ifdef _TASK_TICKLESS
1689 // if there is a task ready to run right now - we have to have another scheduler pass right after this one
1690 // this is especially important for the tasks that run out of iterations and should be disabled during next pass
1691 nrd |= _TASK_NEXTRUN_IMMEDIATE; // next run immediate
1692#endif
1693
1694 if ( iCurrent->iIterations > 0 ) iCurrent->iIterations = iCurrent->iIterations - 1; // do not decrement (-1) being a signal of never-ending task
1695 iCurrent->iRunCounter = iCurrent->iRunCounter + 1;
1696#ifdef _TASK_SCHEDULING_OPTIONS
1697 switch (iCurrent->iOption) {
1698 case TASK_INTERVAL:
1699 iCurrent->iPreviousMillis = m;
1700 break;
1701
1702 case TASK_SCHEDULE_NC:
1703 iCurrent->iPreviousMillis = iCurrent->iPreviousMillis + iCurrent->iDelay;
1704 {
1705 long ov = (long) ( iCurrent->iPreviousMillis + i - m );
1706 if ( ov < 0 ) {
1707 long ii = (i == 0) ? 1 : i; // avoid division by zero
1708 iCurrent->iPreviousMillis = iCurrent->iPreviousMillis + ((m - iCurrent->iPreviousMillis) / ii) * ii;
1709 }
1710 }
1711 break;
1712
1713 default:
1714 iCurrent->iPreviousMillis = iCurrent->iPreviousMillis + iCurrent->iDelay;
1715 }
1716#else
1717 iCurrent->iPreviousMillis = iCurrent->iPreviousMillis + iCurrent->iDelay;
1718#endif // _TASK_SCHEDULING_OPTIONS
1719
1720#ifdef _TASK_TIMECRITICAL
1721 // Updated_previous+current interval should put us into the future, so iOverrun should be positive or zero.
1722 // If negative - the task is behind (next execution time is already in the past)
1723 unsigned long p = iCurrent->iPreviousMillis;
1724 iCurrent->iOverrun = (long) ( p + i - m );
1725 iCurrent->iStartDelay = (long) ( m - p );
1726#endif // _TASK_TIMECRITICAL
1727
1728 iCurrent->iDelay = i;
1729
1730#if defined(_TASK_TIMECRITICAL)
1731 tTaskStart = _task_micros();
1732#endif // _TASK_TIMECRITICAL
1733
1734#ifdef _TASK_OO_CALLBACKS
1735 idleRun = !iCurrent->Callback();
1736#else
1737 if ( iCurrent->iCallback ) {
1738 iCurrent->iCallback();
1739 idleRun = false;
1740 iInvokedTasks++;
1741 }
1742#endif // _TASK_OO_CALLBACKS
1743
1744#if defined(_TASK_TIMECRITICAL)
1745 tTaskFinish = _task_micros();
1746#endif // _TASK_TIMECRITICAL
1747
1748 }
1749#ifdef _TASK_SELF_DESTRUCT
1750 else if ( iCurrent->iStatus.sd_request ) delete iCurrent;
1751#endif // #ifdef _TASK_SELF_DESTRUCT
1752 } while (0); //guaranteed single run - allows use of "break" to exit
1753
1754 iCurrent = nextTask;
1755
1756#ifdef _TASK_TIMECRITICAL
1757 iCPUCycle += ( (_task_micros() - tPassStart) - (tTaskFinish - tTaskStart) );
1758#endif // _TASK_TIMECRITICAL
1759
1760// _TASK_DO_NOT_YIELD is allowed for everyone - you should know what you are doing
1761// #if defined (ARDUINO_ARCH_ESP8266) || defined (ARDUINO_ARCH_ESP32)
1762#if !defined(_TASK_DO_NOT_YIELD)
1763 _task_yield();
1764#endif // _TASK_DO_NOT_YIELD
1765// #endif // ARDUINO_ARCH_ESPxx
1766 }
1767
1768#ifdef _TASK_SLEEP_ON_IDLE_RUN
1769 tFinish = _task_micros(); // Scheduling pass end time in microseconds.
1770#endif
1771
1772#ifdef _TASK_TICKLESS
1773 iNextRun = 0; // next iteration should be immediate by default
1774 // if the pass was "idle" and there are tasks scheduled
1775 do {
1776 if ( !idleRun ) break;
1777 if ( (nrd & _TASK_NEXTRUN_IMMEDIATE) ) break;
1778 if ( nrd == _TASK_NEXTRUN_UNDEFINED ) break;
1779 m = __TASK_TIME_FUNCTION();
1780 if ( nr <= m) break;
1781 iNextRun = ( nr - m );
1782 } while (0);
1783#endif
1784
1785#ifdef _TASK_SLEEP_ON_IDLE_RUN
1786
1787 if (idleRun && iAllowSleep) {
1788 if ( iSleepScheduler == this ) { // only one scheduler should make the MC go to sleep.
1789 if ( iSleepMethod != NULL ) {
1790
1791#ifdef _TASK_TIMECRITICAL
1792 tIdleStart = _task_micros();
1793#endif // _TASK_TIMECRITICAL
1794
1795 (*iSleepMethod)( tFinish-tStart );
1796
1797#ifdef _TASK_TIMECRITICAL
1798 iCPUIdle += (_task_micros() - tIdleStart);
1799#endif // _TASK_TIMECRITICAL
1800 }
1801 }
1802 }
1803
1804
1805
1806#endif // _TASK_SLEEP_ON_IDLE_RUN
1807
1808 return (idleRun);
1809}
